SELECT 
    TO_CHAR(n.time_stamp, 'mm/dd/yyyy HH24:MI:SS') 
    || ' ' || 
    REGEXP_REPLACE(n.note, '[^A-Za-z0-9 ]', ' ') AS cleaned_output
FROM TLM_BDR_NOTE_LINK nl
JOIN TLM_BDR_NOTE n 
  ON nl.note_id = n.id
WHERE nl.LINKED_ENTITY_TYPE = 8018
  AND nl.LINKED_ENTITY_KEY  = '30715957823'
ORDER BY n.time_stamp DESC;
