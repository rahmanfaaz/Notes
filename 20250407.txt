SELECT
  sql_id,
  ROUND(sharable_mem / (1024 * 1024), 2) AS sharable_mem_mb,
  ROUND(persistent_mem / (1024 * 1024), 2) AS persistent_mem_mb,
  ROUND(runtime_mem / (1024 * 1024), 2) AS runtime_mem_mb,
  ROUND((sharable_mem + persistent_mem + runtime_mem) / (1024 * 1024), 2) AS total_mem_mb,
  parsing_schema_name,
  sql_text
FROM v$sql
WHERE UPPER(sql_text) LIKE '%YOUR_KEYWORD%' -- <- filter for your query
AND ROWNUM <= 5
ORDER BY total_mem_mb DESC;
